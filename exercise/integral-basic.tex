\begin{exercise}{Határozzuk meg az alábbi integrál értékét!}
  \[
    \int_0^3 \int_2^y 8xy \, \mathrm dx \, \mathrm dy
  \]

  \exsol{%
    \[
      \int_0^3 \int_2^y 8xy \, \mathrm dx \, \mathrm dy =
      \int_0^3  \Big[ 4x^2y  \Big]_2^y \, \mathrm dy =
      \int_0^3  4y^3 - 16y \, \mathrm dy =
      \Big[ y^4 - 8y^2 \Big]_0^3 =
      81 - 72 =
      9
    \]
  }
\end{exercise}
